async function githubBackupLoad(){ const token = document.getElementById("ghToken")?.value || ""; const gistId = document.getElementById("ghGistId")?.value || ""; if(!token || !gistId){ alert("Informe token e gist ID."); return; } try{ const res = await fetch(`https://api.github.com/gists/${gistId}`, { headers: { "Authorization": `Bearer ${token}` } }); const data = await res.json(); const file = Object.values(data.files || {})[0]; const parsed = JSON.parse(file.content); const restored = parsed.payload || parsed; player = restored.player; enemy = restored.enemy; battleActive = restored.battleActive; battleLog = restored.battleLog; currentScreen = restored.currentScreen; saveGame(); render(); alert("Save carregado!"); } catch(e){ alert("Erro ao carregar save."); } }
